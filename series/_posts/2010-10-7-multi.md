  var bollinger = fc.indicators.bollingerBands();

  var multi = fc.series.multi()
      .series([gridlines, candlestick, bollinger])
      .xScale(xScale)
      .yScale(yScale);

  container.append('g')
      .datum(data)
      .call(multi);
